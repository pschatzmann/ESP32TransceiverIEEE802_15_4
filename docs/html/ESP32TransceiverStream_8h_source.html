319 bool receive() {
320 frame_data_t packet; // Temporary storage for received frame data
321 if (is_open_frame) {
322 // We have a pending frame that we haven't processed yet
323 if (frame.payloadLen > rx_buffer.availableForWrite()) {
324 delay(10);
325 return false;
326 }
327 // Store payload in receive buffer
328 rx_buffer.writeArray(frame.payload, frame.payloadLen);
329 is_open_frame = false; // Mark frame as processed
330 return true;
331 }
332
333 // get next frame
334 size_t read_bytes = 0;
335 read_bytes = xMessageBufferReceive(transceiver.getMessageBuffer(), &packet,
336 sizeof(frame_data_t), pdMS_TO_TICKS(20));
337 if (read_bytes != sizeof(frame_data_t)) {
338 if (read_bytes != 0) {
339 ESP_LOGE(TAG, "Invalid packet size received: %d", read_bytes);
340 }
341 return false;
342 }
343
344 // Parse frame
345 if (!frame.parse(packet.frame, false)) {
346 ESP_LOGE(TAG, "Failed to parse frame");
347 return false;
348 }
349
350 ESP_LOGI(TAG, "Received frame: len=%d, seq=%d", frame.payloadLen,
351 frame.sequenceNumber);
352
353 // Sequence number check (after successful parse, before buffer handling)
354 if (isSequenceNumbers()) {
355 int seq = frame.sequenceNumber;
356 if (last_seq != -1) {
357 int expected = (last_seq + 1) % 256;
358 if (seq == last_seq) {
359 ESP_LOGI(TAG, "Retransmission ignored: seq %d", seq);
360 return false; // Ignore duplicate
361 } else if (seq != expected) {
362 ESP_LOGI(TAG, "Frame sequence skipped: expected %d, got %d", expected,
363 seq);
364 // Accept the new frame, but log the skip
365 }
366 }
367 last_seq = seq;
368 }
369
370 if (frame.payloadLen > rx_buffer.availableForWrite()) {
371 // will be made availabe with next call
372 ESP_LOGD(TAG, "Received frame payload too large for buffer: %d bytes",
373 frame.payloadLen);
374 is_open_frame = true;
375 return false;
376 }
377
378 // Store payload in receive buffer
379 rx_buffer.writeArray(frame.payload, frame.payloadLen);
380 delay(5);
381
382 return true;
383 }

390 void sendWithConfirmations() {
391 uint8_t tmp[tx_buffer.available()];
392 int len = tx_buffer.readArray(tmp, tx_buffer.available());
393 int retry = send_retry_count;
394 // send frame
395 int attempt = 0;
396 do {
397 send_confirmation_state = WAITING_FOR_CONFIRMATION;
398 ESP_LOGD(TAG, "Attempt %d: Sending frame, len: %d", attempt, len);
399 if (!transceiver.send(tmp, len)) {
400 ESP_LOGE(TAG, "Failed to send frame: size %d", len);
401 send_confirmation_state = CONFIRMATION_ERROR;
402 }
403 // wait for confirmations
404 uint32_t timeout =
405 millis() + getAckTimeoutUs() / 1000 + 100; // Add some margin
406 while (send_confirmation_state == WAITING_FOR_CONFIRMATION &&
407 millis() < timeout) {
408 delay(10);
409 }
410
411 // on error retry sending the same frame
412 switch (send_confirmation_state) {
413 case CONFIRMATION_ERROR: {
414 ESP_LOGI(TAG, "Send failed with rc=%d, retrying...", last_tx_error);
415 retry--;
416 if (retry <= 0) {
417 transceiver.incrementSequenceNumber(1);
418 return;
419 }
420 delay(send_delay_ms); // Short delay before retrying if needed
421 break;
422 }
423 case CONFIRMATION_RECEIVED: {
424 transceiver.incrementSequenceNumber(1);
425 break;
426 }
427 default:
428 retry--;
429 delay(send_delay_ms); // Short delay before retrying if needed
430 break;
431 }
432 ++attempt;
433 } while (send_confirmation_state == CONFIRMATION_ERROR);
434 delay(send_delay_ms); // Short delay before retrying if needed
435 }
