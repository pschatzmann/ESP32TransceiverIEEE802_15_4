253 bool receive() {
254 frame_data_t packet; // Temporary storage for received frame data
255 if (is_open_frame) {
256 // We have a pending frame that we haven't processed yet
257 if (frame.payloadLen > rx_buffer.availableForWrite()) {
258 delay(10);
259 return false;
260 }
261 // Store payload in receive buffer
262 rx_buffer.writeArray(frame.payload, frame.payloadLen);
263 is_open_frame = false; // Mark frame as processed
264 return true;
265 }
266
267 // get next frame
268 size_t read_bytes = 0;
269 read_bytes = xMessageBufferReceive(transceiver.getMessageBuffer(), &packet,
270 sizeof(frame_data_t), pdMS_TO_TICKS(20));
271 if (read_bytes != sizeof(frame_data_t)) {
272 if (read_bytes != 0) {
273 ESP_LOGE(TAG, "Invalid packet size received: %d", read_bytes);
274 }
275 return false;
276 }
277
278 // Parse frame
279 if (!frame.parse(packet.frame, false)) {
280 ESP_LOGE(TAG, "Failed to parse frame");
281 return false;
282 }
283
284 ESP_LOGI(TAG, "Received frame: len=%d, seq=%d", frame.payloadLen,
285 frame.sequenceNumber);
286
287 // Sequence number check (after successful parse, before buffer handling)
288 if (isSequenceNumbers()) {
289 int seq = frame.sequenceNumber;
290 if (last_seq != -1) {
291 int expected = (last_seq + 1) % 256;
292 if (seq == last_seq) {
293 ESP_LOGI(TAG, "Retransmission ignored: seq %d", seq);
294 return false; // Ignore duplicate
295 } else if (seq != expected) {
296 ESP_LOGI(TAG, "Frame sequence skipped: expected %d, got %d", expected,
297 seq);
298 // Accept the new frame, but log the skip
299 }
300 }
301 last_seq = seq;
302 }
303
304 if (frame.payloadLen > rx_buffer.availableForWrite()) {
305 // will be made availabe with next call
306 ESP_LOGD(TAG, "Received frame payload too large for buffer: %d bytes",
307 frame.payloadLen);
308 is_open_frame = true;
309 return false;
310 }
311
312 // Store payload in receive buffer
313 rx_buffer.writeArray(frame.payload, frame.payloadLen);
314 delay(5);
315
316 return true;
317 }

324 void sendWithConfirmations() {
325 uint8_t tmp[tx_buffer.available()];
326 int len = tx_buffer.readArray(tmp, tx_buffer.available());
327 int retry = send_retry_count;
328 // send frame
329 int attempt = 0;
330 do {
331 send_confirmation_state = WAITING_FOR_CONFIRMATION;
332 ESP_LOGD(TAG, "Attempt %d: Sending frame, len: %d", attempt, len);
333 if (!transceiver.send(tmp, len)) {
334 ESP_LOGE(TAG, "Failed to send frame: size %d", len);
335 send_confirmation_state = CONFIRMATION_ERROR;
336 }
337 // wait for confirmations
338 uint32_t timeout =
339 millis() + ackTimeoutUs() / 1000 + 100; // Add some margin
340 while (send_confirmation_state == WAITING_FOR_CONFIRMATION &&
341 millis() < timeout) {
342 delay(10);
343 }
344
345 // on error retry sending the same frame
346 switch (send_confirmation_state) {
347 case CONFIRMATION_ERROR: {
348 ESP_LOGI(TAG, "Send failed, retrying...");
349 retry--;
350 if (retry <= 0) {
351 transceiver.incrementSequenceNumber(1);
352 return;
353 }
354 delay(send_delay_ms); // Short delay before retrying if needed
355 break;
356 }
357 case CONFIRMATION_RECEIVED: {
358 transceiver.incrementSequenceNumber(1);
359 break;
360 }
361 default:
362 retry--;
363 delay(send_delay_ms); // Short delay before retrying if needed
364 break;
365 }
366 ++attempt;
367 } while (send_confirmation_state == CONFIRMATION_ERROR);
368 }

374 void sendWithoutConfirmations() {
375 uint8_t tmp[tx_buffer.available()];
376 int len = tx_buffer.readArray(tmp, tx_buffer.available());
377 ESP_LOGD(TAG, "Sending frame, len: %d", len);
378 if (transceiver.send(tmp, len)) {

380 } else {
381 ESP_LOGE(TAG, "Failed to send frame: size %d", len);
382 }
383 delay(send_delay_ms);
384 }
