358 bool receive() {
359 frame_data_t packet; // Temporary storage for received frame data
360 if (is_open_frame) {
361 // We have a pending frame that we haven't processed yet
362 if (frame.payloadLen > rx_buffer.availableForWrite()) {
363 delay(10);
364 return false;
365 }
366 // Store payload in receive buffer
367 rx_buffer.writeArray(frame.payload, frame.payloadLen);
368 is_open_frame = false; // Mark frame as processed
369 return true;
370 }
371
372 // get next frame
373 size_t read_bytes = 0;
374 read_bytes =
375 xMessageBufferReceive(p_transceiver->getMessageBuffer(), &packet,
376 sizeof(frame_data_t), pdMS_TO_TICKS(20));
377 if (read_bytes != sizeof(frame_data_t)) {
378 if (read_bytes != 0) {
379 ESP_LOGE(TAG, "Invalid packet size received: %d", read_bytes);
380 }
381 return false;
382 }
383
384 // Parse frame
385 if (!frame.parse(packet.frame, false)) {
386 ESP_LOGE(TAG, "Failed to parse frame");
387 return false;
388 }
389
390 ESP_LOGI(TAG, "Received frame: len=%d, seq=%d", frame.payloadLen,
391 frame.sequenceNumber);
392
393 // Sequence number check (after successful parse, before buffer handling)
394 if (isSequenceNumbers()) {
395 int seq = frame.sequenceNumber;
396 if (last_seq != -1) {
397 int expected = (last_seq + 1) % 256;
398 if (seq == last_seq) {
399 ESP_LOGI(TAG, "Retransmission ignored: seq %d", seq);
400 return false; // Ignore duplicate
401 } else if (seq != expected) {
402 ESP_LOGI(TAG, "Frame sequence skipped: expected %d, got %d", expected,
403 seq);
404 // Accept the new frame, but log the skip
405 }
406 }
407 last_seq = seq;
408 }
409
410 if (frame.payloadLen > rx_buffer.availableForWrite()) {
411 // will be made availabe with next call
412 ESP_LOGD(TAG, "Received frame payload too large for buffer: %d bytes",
413 frame.payloadLen);
414 is_open_frame = true;
415 return false;
416 }
417
418 // Store payload in receive buffer
419 rx_buffer.writeArray(frame.payload, frame.payloadLen);
420 delay(5);
421
422 return true;
423 }

430 void sendWithConfirmations() {
431 uint8_t tmp[tx_buffer.available()];
432 int len = tx_buffer.readArray(tmp, tx_buffer.available());
433 int retry = send_retry_count;
434 // send frame
435 int attempt = 0;
436 do {
437 send_confirmation_state = WAITING_FOR_CONFIRMATION;
438 ESP_LOGD(TAG, "Attempt %d: Sending frame, len: %d", attempt, len);
439 if (!p_transceiver->send(tmp, len)) {
440 ESP_LOGE(TAG, "Failed to send frame: size %d", len);
441 send_confirmation_state = CONFIRMATION_ERROR;
442 }
443 // wait for confirmations
444 uint32_t timeout =
445 millis() + getAckTimeoutUs() / 1000 + 100; // Add some margin
446 while (send_confirmation_state == WAITING_FOR_CONFIRMATION &&
447 millis() < timeout) {
448 delay(10);
449 }
450
451 // on error retry sending the same frame
452 switch (send_confirmation_state) {
453 case CONFIRMATION_ERROR: {
454 ESP_LOGI(TAG, "Send failed with rc=%d, retrying...", last_tx_error);
455 retry--;
456 if (retry <= 0) {
457 p_transceiver->incrementSequenceNumber(1);
458 return;
459 }
460 delay(send_delay_ms); // Short delay before retrying if needed
461 break;
462 }
463 case CONFIRMATION_RECEIVED: {
464 p_transceiver->incrementSequenceNumber(1);
465 break;
466 }
467 default:
468 retry--;
469 delay(send_delay_ms); // Short delay before retrying if needed
470 break;
471 }
472 ++attempt;
473 } while (send_confirmation_state == CONFIRMATION_ERROR);
474 delay(send_delay_ms); // Short delay before retrying if needed
475 }
