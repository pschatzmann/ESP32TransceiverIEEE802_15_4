295 bool receive() {
296 frame_data_t packet; // Temporary storage for received frame data
297 if (is_open_frame) {
298 // We have a pending frame that we haven't processed yet
299 if (frame.payloadLen > rx_buffer.availableForWrite()) {
300 delay(10);
301 return false;
302 }
303 // Store payload in receive buffer
304 rx_buffer.writeArray(frame.payload, frame.payloadLen);
305 is_open_frame = false; // Mark frame as processed
306 return true;
307 }
308
309 // get next frame
310 size_t read_bytes = 0;
311 read_bytes = xMessageBufferReceive(transceiver.getMessageBuffer(), &packet,
312 sizeof(frame_data_t), pdMS_TO_TICKS(20));
313 if (read_bytes != sizeof(frame_data_t)) {
314 if (read_bytes != 0) {
315 ESP_LOGE(TAG, "Invalid packet size received: %d", read_bytes);
316 }
317 return false;
318 }
319
320 // Parse frame
321 if (!frame.parse(packet.frame, false)) {
322 ESP_LOGE(TAG, "Failed to parse frame");
323 return false;
324 }
325
326 ESP_LOGI(TAG, "Received frame: len=%d, seq=%d", frame.payloadLen,
327 frame.sequenceNumber);
328
329 // Sequence number check (after successful parse, before buffer handling)
330 if (isSequenceNumbers()) {
331 int seq = frame.sequenceNumber;
332 if (last_seq != -1) {
333 int expected = (last_seq + 1) % 256;
334 if (seq == last_seq) {
335 ESP_LOGI(TAG, "Retransmission ignored: seq %d", seq);
336 return false; // Ignore duplicate
337 } else if (seq != expected) {
338 ESP_LOGI(TAG, "Frame sequence skipped: expected %d, got %d", expected,
339 seq);
340 // Accept the new frame, but log the skip
341 }
342 }
343 last_seq = seq;
344 }
345
346 if (frame.payloadLen > rx_buffer.availableForWrite()) {
347 // will be made availabe with next call
348 ESP_LOGD(TAG, "Received frame payload too large for buffer: %d bytes",
349 frame.payloadLen);
350 is_open_frame = true;
351 return false;
352 }
353
354 // Store payload in receive buffer
355 rx_buffer.writeArray(frame.payload, frame.payloadLen);
356 delay(5);
357
358 return true;
359 }

366 void sendWithConfirmations() {
367 uint8_t tmp[tx_buffer.available()];
368 int len = tx_buffer.readArray(tmp, tx_buffer.available());
369 int retry = send_retry_count;
370 // send frame
371 int attempt = 0;
372 do {
373 send_confirmation_state = WAITING_FOR_CONFIRMATION;
374 ESP_LOGD(TAG, "Attempt %d: Sending frame, len: %d", attempt, len);
375 if (!transceiver.send(tmp, len)) {
376 ESP_LOGE(TAG, "Failed to send frame: size %d", len);
377 send_confirmation_state = CONFIRMATION_ERROR;
378 }
379 // wait for confirmations
380 uint32_t timeout =
381 millis() + getAckTimeoutUs() / 1000 + 100; // Add some margin
382 while (send_confirmation_state == WAITING_FOR_CONFIRMATION &&
383 millis() < timeout) {
384 delay(10);
385 }
386
387 // on error retry sending the same frame
388 switch (send_confirmation_state) {
389 case CONFIRMATION_ERROR: {
390 ESP_LOGI(TAG, "Send failed with rc=%d, retrying...", last_tx_error);
391 retry--;
392 if (retry <= 0) {
393 transceiver.incrementSequenceNumber(1);
394 return;
395 }
396 delay(send_delay_ms); // Short delay before retrying if needed
397 break;
398 }
399 case CONFIRMATION_RECEIVED: {
400 transceiver.incrementSequenceNumber(1);
401 break;
402 }
403 default:
404 retry--;
405 delay(send_delay_ms); // Short delay before retrying if needed
406 break;
407 }
408 ++attempt;
409 } while (send_confirmation_state == CONFIRMATION_ERROR);
410 delay(send_delay_ms); // Short delay before retrying if needed
411 }
