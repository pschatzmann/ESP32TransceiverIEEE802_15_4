424 bool receive() {
425 frame_data_t packet; // Temporary storage for received frame data
426 if (is_open_frame) {
427 // We have a pending frame that we haven't processed yet
428 if (frame.payloadLen > rx_buffer.availableForWrite()) {
429 delay(10);
430 return false;
431 }
432 // Store payload in receive buffer
433 rx_buffer.writeArray(frame.payload, frame.payloadLen);
434 is_open_frame = false; // Mark frame as processed
435 return true;
436 }
437
438 // get next frame
439 size_t read_bytes = 0;
440 read_bytes =
441 xMessageBufferReceive(p_transceiver->getMessageBuffer(), &packet,
442 sizeof(frame_data_t), pdMS_TO_TICKS(20));
443 if (read_bytes != sizeof(frame_data_t)) {
444 if (read_bytes != 0) {
445 ESP_LOGE(TAG, "Invalid packet size received: %d", read_bytes);
446 }
447 return false;
448 }
449
450 // Parse frame
451 if (!frame.parse(packet.frame, false)) {
452 ESP_LOGE(TAG, "Failed to parse frame");
453 return false;
454 }
455
456 ESP_LOGI(TAG, "Received frame: len=%d, seq=%d", frame.payloadLen,
457 frame.sequenceNumber);
458
459 // Sequence number check (after successful parse, before buffer handling)
460 if (isSequenceNumbers()) {
461 int seq = frame.sequenceNumber;
462 if (last_seq != -1) {
463 int expected = (last_seq + 1) % 256;
464 if (seq == last_seq) {
465 ESP_LOGI(TAG, "Retransmission ignored: seq %d", seq);
466 return false; // Ignore duplicate
467 } else if (seq != expected) {
468 ESP_LOGI(TAG, "Frame sequence skipped: expected %d, got %d", expected,
469 seq);
470 // Accept the new frame, but log the skip
471 }
472 }
473 last_seq = seq;
474 }
475
476 if (frame.payloadLen > rx_buffer.availableForWrite()) {
477 // will be made availabe with next call
478 ESP_LOGD(TAG, "Received frame payload too large for buffer: %d bytes",
479 frame.payloadLen);
480 is_open_frame = true;
481 return false;
482 }
483
484 // Store payload in receive buffer
485 rx_buffer.writeArray(frame.payload, frame.payloadLen);
486 delay(5);
487
488 return true;
489 }

496 void sendWithConfirmations() {
497 uint8_t tmp[tx_buffer.available()];
498 int len = tx_buffer.readArray(tmp, tx_buffer.available());
499 int retry = send_retry_count;
500 // send frame
501 int attempt = 0;
502 do {
503 send_confirmation_state = WAITING_FOR_CONFIRMATION;
504 ESP_LOGD(TAG, "Attempt %d: Sending frame, len: %d", attempt, len);
505 if (!p_transceiver->send(tmp, len)) {
506 ESP_LOGE(TAG, "Failed to send frame: size %d", len);
507 send_confirmation_state = CONFIRMATION_ERROR;
508 }
509 // wait for confirmations
510 uint32_t timeout =
511 millis() + getAckTimeoutUs() / 1000 + 100; // Add some margin
512 while (send_confirmation_state == WAITING_FOR_CONFIRMATION &&
513 millis() < timeout) {
514 delay(10);
515 }
516
517 // on error retry sending the same frame
518 switch (send_confirmation_state) {
519 case CONFIRMATION_ERROR: {
520 ESP_LOGI(TAG, "Send failed with rc=%d, retrying...", last_tx_error);
521 retry--;
522 if (retry <= 0) {
523 p_transceiver->incrementSequenceNumber(1);
524 return;
525 }
526 delay(send_delay_ms); // Short delay before retrying if needed
527 break;
528 }
529 case CONFIRMATION_RECEIVED: {
530 p_transceiver->incrementSequenceNumber(1);
531 break;
532 }
533 default:
534 retry--;
535 delay(send_delay_ms); // Short delay before retrying if needed
536 break;
537 }
538 ++attempt;
539 } while (send_confirmation_state == CONFIRMATION_ERROR);
540 delay(send_delay_ms); // Short delay before retrying if needed
541 }
